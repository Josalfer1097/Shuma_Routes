-- Extensiones necesarias
create extension if not exists "uuid-ossp";

-- Bodegas de salida/regreso
create table public.depots (
  id uuid primary key default uuid_generate_v4(),
  name text not null,
  address text,
  lat double precision not null,
  lng double precision not null,
  active boolean default true,
  created_at timestamptz default now()
);

-- Catálogo de vehículos
create table public.vehicles (
  id uuid primary key default uuid_generate_v4(),
  plate text not null unique,
  type text not null check (type in ('van','truck_small','truck_large')),
  max_load int not null default 4,
  active boolean default true,
  created_at timestamptz default now()
);

-- Catálogo de choferes
create table public.drivers (
  id uuid primary key default uuid_generate_v4(),
  name text not null,
  employee_id text unique,
  phone text,
  vehicle_id uuid references public.vehicles(id) on delete set null,
  active boolean default true,
  created_at timestamptz default now()
);

-- Cabecera de cada ruta de entrega
create table public.routes (
  id uuid primary key default uuid_generate_v4(),
  date date not null default current_date,
  depot_id uuid references public.depots(id) on delete set null,
  return_depot_id uuid references public.depots(id) on delete set null,
  departure_time time not null,
  status text not null default 'draft'
    check (status in ('draft','optimized','in_progress','completed','cancelled')),
  total_deliveries int default 0,
  total_drivers int default 0,
  created_by text,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- Choferes asignados a cada ruta
create table public.route_drivers (
  id uuid primary key default uuid_generate_v4(),
  route_id uuid not null references public.routes(id) on delete cascade,
  driver_id uuid not null references public.drivers(id) on delete cascade,
  vehicle_id uuid references public.vehicles(id) on delete set null,
  departure_time time,
  color text,
  route_order int,
  total_km double precision,
  total_time_min int,
  created_at timestamptz default now(),
  unique(route_id, driver_id)
);

-- Entregas de cada ruta
create table public.deliveries (
  id uuid primary key default uuid_generate_v4(),
  route_id uuid not null references public.routes(id) on delete cascade,
  route_driver_id uuid references public.route_drivers(id) on delete set null,
  invoice text not null,
  client_name text,
  address text not null,
  lat double precision,
  lng double precision,
  geocoded boolean default false,
  stop_order int,
  status text not null default 'pending'
    check (status in ('pending','in_route','delivered','failed')),
  notes text,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- Log de eventos por entrega
create table public.delivery_events (
  id uuid primary key default uuid_generate_v4(),
  delivery_id uuid not null references public.deliveries(id) on delete cascade,
  event_type text not null
    check (event_type in ('delivered','failed','photo','signature','panic','note')),
  lat double precision,
  lng double precision,
  photo_url text,
  signature_url text,
  notes text,
  created_at timestamptz default now()
);

-- Índices para queries frecuentes
create index idx_routes_date on public.routes(date desc);
create index idx_routes_status on public.routes(status);
create index idx_deliveries_route on public.deliveries(route_id);
create index idx_deliveries_status on public.deliveries(status);
create index idx_delivery_events_delivery on public.delivery_events(delivery_id);
create index idx_route_drivers_route on public.route_drivers(route_id);

-- Trigger para updated_at automático
create or replace function public.handle_updated_at()
returns trigger as $$
begin
  new.updated_at = now();
  return new;
end;
$$ language plpgsql;

create trigger routes_updated_at
  before update on public.routes
  for each row execute function public.handle_updated_at();

create trigger deliveries_updated_at
  before update on public.deliveries
  for each row execute function public.handle_updated_at();

-- ─── 2. BITÁCORA DE MOVIMIENTOS ───────────────

create table public.audit_log (
  id uuid primary key default uuid_generate_v4(),
  action text not null,
  -- Acciones posibles:
  -- Auth:     login_success, login_failed, logout
  -- Rutas:    route_created, route_optimized, route_reoptimized,
  --           route_completed, route_cancelled
  -- Entregas: delivery_moved, delivery_completed, delivery_failed
  -- PDF:      pdf_downloaded
  -- Catálogo: driver_created, driver_updated, driver_deleted,
  --           vehicle_created, vehicle_updated
  -- Pánico:   panic_triggered
  entity text,        -- 'route', 'delivery', 'driver', 'vehicle', 'session'
  entity_id text,     -- UUID del registro afectado
  user_name text,     -- username de quien hizo la acción
  user_role text,     -- rol en el momento de la acción
  ip_address text,
  user_agent text,
  metadata jsonb,     -- datos extra según la acción
  created_at timestamptz default now()
);

create index idx_audit_action   on public.audit_log(action);
create index idx_audit_user     on public.audit_log(user_name);
create index idx_audit_created  on public.audit_log(created_at desc);
create index idx_audit_entity   on public.audit_log(entity, entity_id);
create index idx_audit_role     on public.audit_log(user_role);

-- RLS — solo el backend escribe y lee, nadie puede borrar
alter table public.audit_log enable row level security;

create policy "backend write audit"
  on public.audit_log for insert to service_role with check (true);

create policy "backend read audit"
  on public.audit_log for select to service_role using (true);
